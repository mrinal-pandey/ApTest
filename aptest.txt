	username varchar(50),
	arithmetic_reasoning_score INTEGER DEFAULT 0,

	id INTEGER NOT NULL, 
	username VARCHAR(50) NOT NULL, 
	password VARCHAR(255) DEFAULT '' NOT NULL, 
	active BOOLEAN DEFAULT '0', 
	email VARCHAR(255) NOT NULL, 
	bio VARCHAR(500) DEFAULT "", 
	dob VARCHAR(11) DEFAULT '-----------', 
	role BOOLEAN, 
	PRIMARY KEY(id), 
	UNIQUE (username), 
	CHECK(active IN(0, 1)), 
	UNIQUE(email), 
	CHECK(role IN(0, 1))
);
